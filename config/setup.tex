%!TEX root = ../informe.tex

% Package includes and formatting settings -----------
% ----------------------------------------------------

% LaTeX Font encoding - 8-bit fonts
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage{palatino}
\usepackage{titling}
\usepackage{scrhack}
    
% Babel will provide support for spanish
\usepackage[spanish]{babel}

% Setting the palatino font 
\renewcommand{\familydefault}{\sfdefault}
\renewcommand{\sfdefault}{ppl}

% Chapter title, section, subsection formatting
\setkomafont{chapter}{\normalfont\bfseries\huge}
\setkomafont{section}{\normalfont\bfseries\Large}
\setkomafont{subsection}{\normalfont\bfseries\large}

% Continuous numbering of figures/tables over chapters
\usepackage{chngcntr}

% Adjust the style of float's captions, separated from text to improve readabilty
\usepackage[labelfont=bf, labelsep=colon, format=hang, textfont=singlespacing]{caption}

\counterwithout{equation}{chapter}
\counterwithout{figure}{chapter}
\counterwithout{table}{chapter}

% More colors
\usepackage[dvipsnames]{xcolor}
\definecolor{MainColour}{RGB}{0, 53, 107}
    
% Coloured chapter numbers
\renewcommand*{\chapterformat}{
    \textcolor{MainColour}{\thechapter}\enskip
}

% Line spacing
\usepackage{setspace}
\onehalfspacing
\renewcommand{\baselinestretch}{1.5} 

% Functional packages --------------------------------
% ----------------------------------------------------

% LaTeX' own graphics handling
\usepackage{graphicx}

% AMS-LaTeX extensions for mathematical typesetting.
\usepackage{amsmath,amsthm,amsfonts,amssymb,mathrsfs}
\usepackage[sc]{mathpazo}

% Make document internal hyperlinks wherever possible
\usepackage[unicode=true,bookmarks=true,bookmarksnumbered=true,
            bookmarksopen=true,bookmarksopenlevel=1,breaklinks=false,
            pdfborder={0 0 0},backref=false,colorlinks=false]{hyperref}
            
% If-Else commands
\usepackage{etoolbox}

% Other packages -------------------------------------
% ----------------------------------------------------

% Improves general appearance of the text
\usepackage[protrusion=true,expansion=true, kerning]{microtype}
    
% For nicer looking tables
\usepackage{booktabs}
    
% Easy quotations enables \say{}
\usepackage{dirtytalk}

% Other settings -------------------------------------
% ----------------------------------------------------

% Referencing
\newcommand{\secref}[1]{Section~\ref{#1}}
\newcommand{\chapref}[1]{Chapter~\ref{#1}}
\renewcommand{\eqref}[1]{Equation~(\ref{#1})}
\newcommand{\figref}[1]{Figure~\ref{#1}}
\newcommand{\tabref}[1]{Table~\ref{#1}}

% Layout ---------------------------------------------
% ----------------------------------------------------

% Prevents floating objects from being placed ahead of their section    
\let\mySection\section\renewcommand{\section}{\suppressfloats[t]\mySection}
\let\mySubSection\subsection\renewcommand{\subsection}{\suppressfloats[t]\mySubSection}
    
% Math formatting commands ---------------------------
% ----------------------------------------------------

% Define vectors to be bold instead of using an arrow
\renewcommand{\vec}[1]{\mathbf{#1}}
\newcommand{\mat}[1]{\mathbf{#1}}

% Tag equation with name
\newcommand{\eqname}[1]{\tag*{#1}}
	
% Translations ---------------------------------------
% ----------------------------------------------------

% Use "Contenido" instead of "Content" in ToC
\addto\captionsspanish{
    \renewcommand{\contentsname}%
    {Contenido}%
}

% Use "Figura" instead of "Figure"
\addto\captionsspanish{
    \renewcommand{\listfigurename}%
    {Figuras}%
}

% Epigraph -------------------------------------------
% ----------------------------------------------------

% #1 Width
% #2 Text
% #3 Author
\newcommand\epigraph[3]{
\vspace{1em}\hfill{}\begin{minipage}{#1}{\begin{spacing}{0.9}
\small\noindent\textit{``#2''}\end{spacing}
\vspace{1em}
\hfill{}{#3}}\vspace{2em}
\end{minipage}}
